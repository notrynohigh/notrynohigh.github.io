qt.network.ssl: QSslSocket: cannot call unresolved function SSLv23_client_method

转自：

最近在做一个网络音乐播放器时，由于出现qt.network.ssl: QSslSocket: cannot call unresolved function SSLv23_client_method，

而不能播放网络歌曲。

上网搜了半天，都说要在电脑那里安装openssl，然后把C:\OpenSSL-Win64\bin下的libeay32.dll和ssleay32.dll拷贝到D:\Qt\Qt5.4.2\5.4\mingw491_32\bin， 然而并没什么卵用！

我的解决办法是：

Qt的这个目录下：D:\Qt\Qt5.4.2\Tools\mingw491_32\opt\bin。自带libeay32.dll和ssleay32.dll.

只需要把这个目录下的两个dll拷贝到D:\Qt\Qt5.4.2\5.4\mingw491_32\bin，就不会提示qt.network.ssl: QSslSocket: cannot call unresolved function SSLv23_client_method。

35. 搜索插入位置

给定一个排序数组和一个目标值，在数组中找到目标值，并返回其索引。如果目标值不存在于数组中，返回它将会被按顺序插入的位置。

你可以假设数组中无重复元素。

示例 1:

输入: [1,3,5,6], 5 输出: 2

示例 2:

输入: [1,3,5,6], 2 输出: 1

示例 3:

输入: [1,3,5,6], 7 输出: 4

示例 4:

输入: [1,3,5,6], 0 输出: 0

int searchInsert(int* nums, int numsSize, int target) {
int left, right , index;

if(nums == NULL || numsSize == 0)
{
    return 0;
}

if(numsSize == 1)
{
    if(nums\[0\] < target)
    {
        return 1;
    }
    return 0;
}

left = 0;
right = numsSize - 1;
index = left + (right - left) / 2;

while(left <= right)
{
    if(nums\[index\] == target)
    {
        return index;
    }
    if(nums\[index\] > target)
    {
        right = index - 1;
    }
    else
    {
        left = index + 1;
    }
    index = left + (right - left) / 2;
}
return left;

}

34. 在排序数组中查找元素的第一个和最后一个位置

给定一个按照升序排列的整数数组 nums，和一个目标值 target。找出给定目标值在数组中的开始位置和结束位置。

你的算法时间复杂度必须是 O(log n) 级别。

如果数组中不存在目标值，返回 [-1, -1]。

示例 1:

输入: nums = [5,7,7,8,8,10], target = 8 输出: [3,4]

示例 2:

输入: nums = [5,7,7,8,8,10], target = 6 输出: [-1,-1]

/**
* Return an array of size returnSize.
\ Note: The returned array must be malloced, assume caller calls free().
/
int searchRange(int* nums, int numsSize, int target, int* returnSize) {
int *p = malloc(2 * sizeof(int));
int left, right, index;
int left_1, left_2, index_1, right_1, right_2, index_2;
*returnSize = 2;
if(p == NULL)
{
return NULL;
}

p\[0\] = -1;
p\[1\] = -1;

if(nums == NULL || numsSize == 0)
{
    return p;
}

if(numsSize == 1)
{
    if(nums\[0\] == target)
    {
        p\[0\] = 0;
        p\[1\] = 0;
    }
    return p;
}

left = 0;
right = numsSize - 1;
index = left + (right - left) / 2;
while(left <= right)
{
    if(nums\[index\] == target)
    {
        break;
    }
    if(nums\[index\] > target)
    {
        right = index - 1;
    }
    else
    {
        left = index + 1;
    }
    index = left + (right - left) / 2;
}

if(left > right)
{
    return p;
}

left_1 = left;
left_2 = index;
right_1 = index;
right_2 = right;
index\_1 = left\_1 + (right\_1 - left\_1) / 2;
index\_2 = left\_2 + (right\_2 - left\_2) / 2;
while(p\[0\] == -1 || p\[1\] == -1)
{
    if(p\[0\] == -1)
    {
        if(nums\[index_1\] < target)
        {
            if(nums\[index_1 + 1\] == target)
            {
                p\[0\] = index_1 + 1;
            }
            left\_1 = index\_1 + 1;
        }
        else
        {
            if(nums\[index\_1 - 1\] < target || index\_1 == 0)
            {
                p\[0\] = index_1;
            }
            right\_1 = index\_1 - 1;
        }
        index\_1 = left\_1 + (right\_1 - left\_1) / 2;
    }
    if(p\[1\] == -1)
    {
        if(nums\[index_2\] > target)
        {
            if(nums\[index_2 - 1\] == target)
            {
                p\[1\] = index_2 - 1;
            }
            right\_2 = index\_2 - 1;
        }
        else
        {
            if(nums\[index\_2 + 1\] > target || index\_2 == (numsSize - 1))
            {
                p\[1\] = index_2;
            }
            left\_2 = index\_2 + 1;
        } 
        index\_2 = left\_2 + (right\_2 - left\_2) / 2;
    }
}

return p;

}

33. 搜索旋转排序数组

假设按照升序排序的数组在预先未知的某个点上进行了旋转。

( 例如，数组 [0,1,2,4,5,6,7] 可能变为 [4,5,6,7,0,1,2] )。

搜索一个给定的目标值，如果数组中存在这个目标值，则返回它的索引，否则返回 -1 。

你可以假设数组中不存在重复的元素。

你的算法时间复杂度必须是 O(log n) 级别。

示例 1:

输入: nums = [4,5,6,7,0,1,2], target = 0 输出: 4

示例 2:

输入: nums = [4,5,6,7,0,1,2], target = 3 输出: -1

int search(int* nums, int numsSize, int target) {
int direction = 0;
int left = 0, right = numsSize - 1;
if(nums == NULL || numsSize == 0)
{
return -1;
}
if(numsSize == 1)
{
if(nums[0] == target)
{
return 0;
}
return -1;
}

int index = (right - left) / 2;
do
{
    if(nums\[index\] == target)
    {
        return index;
    }
    if(nums\[index\] > nums\[index - 1\] && nums\[index\] > nums\[index + 1\] && index != 0 && index != (numsSize - 1))
    {
        if(target > nums\[index\])
        {
            return -1;
        }
        if(target <= nums\[right\])
        {
            direction = 1;
        }
        else if(target >= nums\[left\])
        {
            direction = 0;
        }
        else
        {
            return -1;
        }
    }
    else if(nums\[index\] < nums\[index - 1\] && nums\[index\] < nums\[index + 1\] && index != 0 && index != (numsSize - 1))
    {
        if(target < nums\[index\] || target > nums\[index - 1\])
        {
            return -1;
        }
        if(target <= nums\[right\])
        {
            direction = 1;
        }
        else if(target >= nums\[left\])
        {
            direction = 0;
        }
        else
        {
            return -1;
        }
    }
    else if(nums\[index\] > target)
    {
        direction = 0;
        if(nums\[left\] > nums\[right\] && nums\[right\] >= target && nums\[index\] >= nums\[left\])
        {
            direction = 1;
        }
    }
    else if(nums\[index\] < target)
    {
        direction = 1;
        if(nums\[right\] < target && nums\[index\] <= nums\[right\])
        {
            direction = 0;
        }
    }

    if(direction)
    {
        left = index + 1;
        index = left + (right - left) / 2;
    }
    else
    {
        right = index - 1;
        index = left + (right - left) / 2;
    }
}while(right >= left);
return -1;

}

32. 最长有效括号

给定一个只包含 ‘(‘ 和 ‘)’ 的字符串，找出最长的包含有效括号的子串的长度。

示例 1:

输入: “(()” 输出: 2 解释: 最长有效括号子串为”()”

示例 2:

输入: “)()())” 输出: 4 解释: 最长有效括号子串为”()()”

int longestValidParentheses(char* s) {
int len_s = strlen(s);
int i, j, max = 0, count_x = 0;
int i_1;
if(s == NULL || len_s < 2)
{
return 0;
}

for(i = 0;i < len_s;i++)
{
    if(s\[i\] == ')')
    {
        for(j = i - 1;j >= 0;j--)
        {
            if(s\[j\] == ')')
            {
                break;
            }
            else if(s\[j\] == '(')
            {
                s\[j\] = '*';
                s\[i\] = '*';
                break;
            }
        }
    }
}

//printf("%s\\r\\n", s);

for(i = 0;i < len_s;i++)
{
    if(s\[i\] == '*' && s\[i - 1\] != '*')
    {
        i_1 = i;
    }
    else if(s\[i\] == '*' && s\[i + 1\] != '*')
    {
        //printf("%d\\r\\n", i);
        count\_x = i - i\_1 + 1;
        if(count_x > max)
        {
            max = count_x;
        }
    }
}
if(s\[len_s - 1\] == '*')
{
    count\_x = len\_s - i_1;
    if(count_x > max)
    {
        max = count_x;
    }
}

return max;

}

31. 下一个排列

实现获取下一个排列的函数，算法需要将给定数字序列重新排列成字典序中下一个更大的排列。

如果不存在下一个更大的排列，则将数字重新排列成最小的排列（即升序排列）。

必须原地修改，只允许使用额外常数空间。

以下是一些例子，输入位于左侧列，其相应输出位于右侧列。
1,2,3 → 1,3,2
3,2,1 → 1,2,3
1,1,5 → 1,5,1

void nextPermutation(int* nums, int numsSize) {
int i = 0;
int j = 0;
int t = 0;
if(nums == NULL || numsSize < 2)
{
return;
}

i = numsSize - 1;
while(nums\[i\] <= nums\[i - 1\] && i > 0)
{
    i--;
}
if(i > 0)
{
    for(j = i + 1;j < numsSize;j++)
    {
        if(nums\[j\] <= nums\[i - 1\])
        {
            break;
        }
    }
    j = j - 1;

    t = nums\[j\];
    nums\[j\] = nums\[i - 1\];
    nums\[i - 1\] = t; 

    for(j = 0;j < (numsSize - i) / 2;j++)
    {
        t = nums\[j + i\];
        nums\[j + i\] = nums\[numsSize - 1 - j\];
        nums\[numsSize - 1 - j\] = t;  
    }
}
else
{
    for(i = 0;i < numsSize / 2;i++)
    {
        t = nums\[i\];
        nums\[i\] = nums\[numsSize - 1 - i\];
        nums\[numsSize - 1 - i\] = t;  
    }
}

}

29. 两数相除

给定两个整数，被除数 dividend 和除数 divisor。将两数相除，要求不使用乘法、除法和 mod 运算符。

返回被除数 dividend 除以除数 divisor 得到的商。

示例 1:

输入: dividend = 10, divisor = 3 输出: 3

示例 2:

输入: dividend = 7, divisor = -3 输出: -2

说明:

  • 被除数和除数均为 32 位有符号整数。
  • 除数不为 0。
  • 假设我们的环境只能存储 32 位有符号整数，其数值范围是 [−231, 231 − 1]。本题中，如果除法结果溢出，则返回 231 − 1。

代码如下：

int divide(int dividend, int divisor) {
int _d = 0x80000000;
int _i = 0x7fffffff;
int result = 1, result_a = 0;
if(dividend == 0)
{
return 0;
}

if(divisor == 1)
{
    return dividend;
}

if(divisor == dividend)
{
    return 1;
}

if(divisor == -1)
{
    if(dividend < 0)
    {
        if(dividend == _d)
        {
            return _i;
        }
        else
            return abs(dividend);
    }
    return (~(dividend) + 1);
}

if(divisor == _d)
{
    return 0;
}

if(dividend != _d)
{
    if(abs(dividend) < abs(divisor))
    {
        return 0;
    }
}

bool flag = true;
bool flag_2 = true;
if((dividend < 0 && divisor > 0) || (dividend > 0 && divisor < 0))
{
    flag = false;
}
int tmp, tmp1, tmp2 = 0;

if(dividend < 0)
{
    flag_2 = false;
}

tmp2 = dividend & 0x1;
dividend = dividend >> 1;

tmp1 = abs(divisor);
if(flag_2)
{
    tmp = tmp1;
}
else tmp = 0 - tmp1;

if(abs(dividend) < abs(divisor))
{
    if(flag) return 1;
    return -1;
}

divisor = tmp;
//printf("%d %d\\r\\n", divisor, dividend);
while(1)
{
    divisor <<= 1;
    if((divisor > dividend && flag\_2) || (divisor < dividend && !flag\_2))
    {
        dividend -= (divisor >> 1);
        if(divisor != (tmp + tmp))
        { 
            //printf("%d %d\\r\\n", divisor, dividend);
            result_a += result;
            result = 1;
            divisor = tmp;
            if(abs(dividend) < abs(divisor))
            {
                break;
            }
            continue;
        }
        result_a += result;
        break;
    }
    result <<= 1;
}

dividend = abs(dividend);

if((dividend + dividend + tmp2) > tmp1) 
{
    result_a++;
}

if(!flag)
{
    result\_a = (~result\_a) + 1;
}
return result_a;

}
