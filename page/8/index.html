#include “stdio.h”

#ifndef NULL
#define NULL ((void *)0)
#endif

int cmp(char *p, char *q)
{
if(p == NULL || q == NULL)
{
return 0;
}

if(\*p == \*q)
{
    if(*p == '\\0')
    {
        return 0;    
    }
    return cmp(p + 1, q + 1);    
}
if(\*p > \*q)
{
    return 1;    
}
return -1;

}

void main()
{
int test;
test = cmp(“abc5”, “abc6”);
printf(“%d”, test);
}

/*
* Insert a new entry between two known consecutive entries.
*
* This is only for internal list manipulation where we know
* the prev/next entries already!

struct list_head *next)

next->prev = new;
new->next = next;
new->prev = prev;
prev->next = new;

}

递归简论

程序=数据结构+算法，所以这本数据结构与算法分析（C描述）刚好合适。开始看看。

递归看起来是一种让代码简洁的方式，但是使用过程还是有点小难。先根据书中一个例子编程体验一下：

设我们有一个整数N并希望把它打印出来，例如编写一个函数print_digit（1234），则依次打印1，2，3，4

对应代码如下：

#include “stdio.h”

typedef unsigned int uint32_t;
typedef unsigned short uin16_t;
typedef unsigned char uint8_t;

void print_digit(uint32_t n)
{
if(n > 10)
{
print_digit(n / 10);
}
printf(“%d “, n % 10);
}

void main()
{
print_digit(1684);
}

看起来代码确实简洁了很多。使用递归还是需要遵循两点：

1、基准情形，总要有某些基准的情形不需要递归就能得出解。

2、不断推进，需要递归求解的情形必须总能朝着基准情形前进

return;
}
base = pbuf[0];

while(left != right)
{
    while(pbuf\[right\] >= base && left < right)
    {
        right--;
    }

    while(pbuf\[left\] <= base && left < right)
    {
        left++;
    }

    if(left < right)
    {
        tmp = pbuf\[left\];
        pbuf\[left\] = pbuf\[right\];
        pbuf\[right\] = tmp;
    }

}
pbuf\[0\] = pbuf\[left\];
pbuf\[left\] = base;

quick_sort(pbuf, left + 1);
quick_sort(&pbuf\[right + 1\], len - right - 1);

}

void printf_table(uint8_t *pbuf, uint32_t len)
{
uint32_t i = 0;
if(pbuf == NULL)
{
return;
}
for(i = 0;i < len;i++)
{
printf(“%d “, pbuf[i]);
}
printf(“\n\r”);
}

void main()
{
quick_sort(table, 10);
printf_table(table, 10);
}

用于MCU项目中，存储一些配置信息。想要实现的目标主要是：{name, value}这样成对存储，查询。针对的存储芯片是SPI FLASH，这类芯片在MCU项目中使用的比较常见而且普遍容量大，拿出几K作为配置信息的存储相当于九牛一毛。

 这个版本只是简单的实现了存储和查询，代码比较粗糙，没有在实际项目中使用的价值。

inc/b_kv.h

src/b_kv.c

port/b\_kv\_port.c b\_kv\_port.h b\_kv\_config.h

配置文件主要针对于MCU平台，同时根据项目的实际需求配置KV对最大条数。

目前这个版本没有回收机制，会导致操作N次后就空间占满无法继续存储或者修改KV对。所以无法再实际项目中使用。

此版本只是为了给自己一些灵感，后续去完善。

nrfutil的使用

nrfutil的使用

命令

  • pkg命令： 这组命令用于生成DFU的包。
    • generate子命令： 生成一个（.zip）的包，后面可以通过手机APP或其它方式进行DFU的升级。
    • 这个命令支持以下的升级包的生成：
      • BL only：支持
      • SD only：支持（SD主版本相同）
      • APP only：支持
      • BL + SD：支持
      • BL + APP：不支持（通过两个包支持）
      • BL + SD + APP：支持
      • SD + APP：支持（SD主版本相同）
    • 参数：
      • –debug-mode： Debug模式切换，使能版本检查。
      • –application TEXT： Application固件文件。
      • –application-version INTEGER： Application固件的版本。
      • –application-version-string TEXT： Application固件的版本字符串。
      • –bootloader TEXT： Bootloader固件文件。
      • –bootloader-version INTEGER： Bootloader固件的版本。
      • –hw-version INTEGER： 硬件版本。
      • –sd-id TEXT： 新的SoftDevice ID
      • –softdevice TEXT： SoftDevice固件的文件。
      • –key-file PATH： 提供的PEM格式的私有（签名的）Key。
      • –help： 显示帮助。
      • –sd-req TEXT： 兼容使用的SoftDevce，可以是一个或多个SoftDevice ID。
        每个ID必须是2位或4位的HEX值，比如”0x12”, “0x1234”。
      • SoftDevice 的ID列表：

  • display子命令： 这个命令显示DFU包（.zip）的信息。

    • e.g.： nrfutil pkg display package.zip

  • dfu子命令： 这个命令可以通过串口、BLE或Thraed进行DFU更新。

    • ble参数： 通过BLE升级。
      • -pkg，–package PATH： DFU包的文件名[必须]。
      • -ic，–conn-ic-id [NRF51|NRF52]： 指明所使用的IC：NRF51或NRF52[必须]。
      • -p，–port TEXT： 指定串口。
      • -n，–name TEXT： 设备名。
      • -a，–address TEXT： 蓝牙地址。
      • -snr，–jlink_snr TEXT： Jlink序列号。
      • -f，–flash_connectivity： 自动烧写connectivity固件。
      • –help： 显示帮助。
    • ble示例： nrfutil dfu ble -ic NRF52 -pkg app_dfu_package.zip -p COM3 -n “MyDevice” -f
    • Thread参数： 通过Thread协议升级，不常用。
    • serial参数： 通过串口进行升级。
      • -pkg，–package PATH： DFU包的文件名[必须]。
      • -p，–port TEXT： 指定串口。
      • -fc，–flow-control BOOLEAN： 使能流控这个值为1。
      • -prn，–packet-receipt-notification INTEGER： Set the packet receipt notification value
      • -b，–baud-rate INTEGER： 设置串口波特率。
      • –help： 显示帮助。
    • serial示例： nrfutil dfu serial -pkg app_dfu_package.zip -p COM3
    • usb_serial示例： 与serial相同。

  • keys子命令： 这个命令生成用于校验DFU包的加密密钥。

    • generate参数： 生成加密密钥，nrfutil keys generate private.pem
    • display参数： 显示加密文件的加密密钥。
      • –key TEXT： （pk | sk）显示公钥（pk，public key），或私钥（sk，private key）。
      • –format TEXT： （hex | code | pem）显示Key的格式：hex格式、C格式、PEM格式。
      • –out_file TEXT： 输出保存的文件名。
      • –help： 显示帮助。

  • settings子命令： 这个命令生成Bootloader DFU settings，必须烧写到Flash的最后一页。

    • generate参数： 生成Bootloader DFU settings，这将生成一个hex文件。
      • –family [NRF51 | NRF52 | NRF52QFAB | NRF52810 | NRF52840]： 指定IC。
      • –application TEXT： 指定Application固件文件。
      • –application-version INTEGER： Application固件的版本。
      • –application-version-string TEXT： Application固件的版本字符串。
      • –bootloader TEXT： Bootloader固件文件。
      • –bootloader-version INTEGER： Bootloader固件的版本。
      • –help： 显示帮助。
    • display参数： 显示Bootloader DFU settings的hex文件的信息。
      • e.g.： nrfutil settings display flash_dump.hex
